private:
    T* arr;
    int capacity;
    int top;

// Copy Constructort
    StackArr(const StackArr<T>& s){
        capacity = s.capacity;
        arr = new T[capacity];
        for(int i = 0; i <= s.top; i++){
            arr[i] = s.arr[i];   // arr is a private memeber
        }
        top = s.top;
    }

Topic - Stack in C++ STL (Standard Template Library)

#Creation:- stack<type> stack_name;
ex:  stack<char> s;

#Insertion:- stack_name.push(data);

ex: s.push('a');

#deletion:- stack_name.pop();

ex: s.pop();

#top_element:- stack_name.top();
ex: s.top();

#empty:- stack_name.empty();

ex: s.empty();

#size:- stack_name.size();

ex: s.size();
